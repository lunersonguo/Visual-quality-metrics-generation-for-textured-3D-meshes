\begin{equation}
e(p, A) = \min_{v^A_i\in A}{d(p, v^A_i)}
\end{equation}
\begin{equation}
H_a(A,B) = \max_{v^A_i\in A}{e(v^A_i, B)}
\end{equation}

\begin{equation}
H_a(A,B) = max\{ H_a(A,B), H_b(A,B) \}
\end{equation}

\begin{equation}
RMS(A,B) = (\sum_{i=1}^n{||v^A_i-v^B_i||^2})^{1/2}
\end{equation}

\begin{equation}
MSE_a(A,B) = \sum_{i=1}^{N_A}}{e(v^A_i, B)^2}
\end{equation}
